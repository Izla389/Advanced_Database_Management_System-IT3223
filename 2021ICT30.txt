CREATE PROCEDURE ReplaceQuantity(
    IN ProductID INT,
    IN NewQuantity INT,
    OUT UpdatedStock INT
)
BEGIN
    -- Update the stock with the new value
    UPDATE PRODUCTS
    SET Quantity = NewQuantity
    WHERE ID = ProductID;

    -- Return the updated stock
    SELECT Quantity INTO UpdatedStock
    FROM PRODUCTS
    WHERE ID = ProductID;
END //

DELIMITER ;

SET @ProdID = 101;
SET @NewQuantity = 50;
SET @UpdatedStock = 0;

CALL ReplaceQuantity(@ProdID, @NewQuantity, @UpdatedStock);
SELECT @UpdatedStock AS NewStock;
+----------+
| NewStock |
+----------+
|        0 |
+----------+

DELIMITER //

CREATE PROCEDURE AdjustQuantity(
    IN ProductID INT,
    IN AdjustBy INT,
    OUT UpdatedStock INT
)
BEGIN
    -- Update the stock by adding the given value (positive or negative)
    UPDATE PRODUCTS
    SET Quantity = Quantity + AdjustBy
    WHERE ID = ProductID;

    -- Return the updated stock
    SELECT Quantity INTO UpdatedStock
    FROM PRODUCTS
    WHERE ID = ProductID;
END //

DELIMITER ;

SET @ProdID = 101;
SET @AdjustBy = -5;  -- negative reduces, positive increases
SET @UpdatedStock = 0;

CALL AdjustQuantity(@ProdID, @AdjustBy, @UpdatedStock);
SELECT @UpdatedStock AS NewStock;
